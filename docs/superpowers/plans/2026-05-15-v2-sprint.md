```python
import uuid
from django.conf import settings
from django.db import models


class Attraction(models.Model):
    CATEGORY_CHOICES = [
        ('TEMPLE', 'Temple'),
        ('BEACH', 'Beach'),
        ('MUSEUM', 'Museum'),
        ('RESTAURANT', 'Restaurant'),
        ('SHOPPING', 'Shopping'),
        ('NATURE', 'Nature'),
        ('EVENT', 'Event'),
    ]

    id = models.UUIDField(primary_key=True, default=uuid.uuid4, editable=False)
    city = models.CharField(max_length=100, db_index=True)
    name = models.CharField(max_length=200)
    category = models.CharField(max_length=20, choices=CATEGORY_CHOICES, db_index=True)
    description = models.TextField(blank=True, default='')
    address = models.CharField(max_length=300, blank=True, default='')
    opening_hrs = models.CharField(max_length=200, blank=True, default='')
    notes = models.TextField(blank=True, default='')
    is_visible = models.BooleanField(default=True, db_index=True)
    sort_order = models.PositiveSmallIntegerField(default=0)
    created_by = models.ForeignKey(
        settings.AUTH_USER_MODEL,
        on_delete=models.SET_NULL,
        null=True, blank=True,
        related_name='attractions_created',
    )
    created_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ['sort_order', 'name']
        verbose_name = 'attraction'
        verbose_name_plural = 'attractions'

    def __str__(self):
        return f"{self.name} ({self.city})"

    @property
    def primary_photo(self):
        return self.photos.filter(is_primary=True).first() or self.photos.first()


class AttractionPhoto(models.Model):
    id = models.UUIDField(primary_key=True, default=uuid.uuid4, editable=False)
    attraction = models.ForeignKey(Attraction, on_delete=models.CASCADE, related_name='photos')
    image = models.ImageField(upload_to='attraction_photos/')
    is_primary = models.BooleanField(default=False)
    sort_order = models.PositiveSmallIntegerField(default=0)

    class Meta:
        ordering = ['sort_order', '-is_primary']
```

- [ ] **Step 1: Add edit_profile view to `accounts/views.py`**

```python
from django.contrib.auth.decorators import login_required
from django.core.cache import cache
from django.http import JsonResponse
from django.views.decorators.http import require_http_methods
import json, re

@login_required(login_url='/accounts/login/page/')
def edit_profile_page(request):
    return render(request, 'accounts/edit_profile.html')


@login_required(login_url='/accounts/login/page/')
@require_http_methods(["POST"])
def update_profile(request):
    """Update name and phone immediately. Email triggers OTP flow."""
    try:
        data = json.loads(request.body)
    except json.JSONDecodeError:
        return JsonResponse({'error': 'Invalid JSON.'}, status=400)

    user = request.user
    field = data.get('field')

    if field == 'name':
        name = data.get('value', '').strip()
        if not name:
            return JsonResponse({'error': 'Name cannot be empty.'}, status=400)
        user.full_name = name
        user.save(update_fields=['full_name'])
        return JsonResponse({'message': 'Name updated.'})

    if field == 'phone':
        phone = data.get('value', '').strip()
        if not re.match(r'^[6-9]\d{9}$', phone):
            return JsonResponse({'error': 'Enter a valid 10-digit Indian phone number.'}, status=400)
        user.phone = phone
        user.save(update_fields=['phone'])
        return JsonResponse({'message': 'Phone updated.'})

    if field == 'email_request':
        new_email = data.get('value', '').strip().lower()
        if not new_email:
            return JsonResponse({'error': 'Email cannot be empty.'}, status=400)
        if User.objects.filter(email=new_email).exclude(pk=user.pk).exists():
            return JsonResponse({'error': 'Email already taken.'}, status=400)
        # Store OTP in cache
        from .utils import create_and_store_otp, send_otp_email
        otp_code = create_and_store_otp(new_email)
        send_otp_email(new_email, otp_code)
        cache.set(f'email_change:{user.pk}:{new_email}', True, timeout=600)
        return JsonResponse({'message': f'Verification code sent to {new_email}.'})

    if field == 'email_verify':
        new_email = data.get('email', '').strip().lower()
        otp = data.get('otp', '').strip()
        cache_key = f'email_change:{user.pk}:{new_email}'
        if not cache.get(cache_key):
            return JsonResponse({'error': 'OTP expired or not initiated.'}, status=400)
        from .utils import verify_otp
        result = verify_otp(new_email, otp)
        if not result['success']:
            return JsonResponse({'error': result['error']}, status=400)
        if User.objects.filter(email=new_email).exclude(pk=user.pk).exists():
            return JsonResponse({'error': 'Email already taken.'}, status=400)
        user.email = new_email
        user.save(update_fields=['email'])
        cache.delete(cache_key)
        return JsonResponse({'message': 'Email updated successfully.'})

    return JsonResponse({'error': 'Unknown field.'}, status=400)
```

```python
class RoleRoutingMiddleware:
    """
    Guest hitting /admin-portal/ → redirect to /accounts/employee/login/
    is_staff hitting /accounts/login/page/ → redirect to /accounts/employee/login/
    """
    def __init__(self, get_response):
        self.get_response = get_response

    def __call__(self, request):
        path = request.path

        # Staff trying to use guest login → send to employee login
        if path.startswith('/accounts/login/page/') and request.user.is_authenticated:
            profile = getattr(request.user, 'userprofile', None)
            if profile and profile.role in ['employee', 'super_admin']:
                from django.urls import reverse
                return redirect(reverse('accounts:employee-login'))

        # Guest trying to access admin portal
        if path.startswith('/admin-portal/') and request.user.is_authenticated:
            profile = getattr(request.user, 'userprofile', None)
            if not profile or profile.role == 'guest':
                return redirect('/accounts/login/page/?next=' + path)

        return self.get_response(request)
```

```python
def award_booking_points(booking_pk):
    from rooms.models import Booking
    from accounts.models import UserProfile
    from .models import LoyaltyConfig, LoyaltyLedger, CampaignRule

    try:
        booking = Booking.objects.select_related('room__property', 'user').get(pk=booking_pk)
    except Booking.DoesNotExist:
        return

    try:
        config = LoyaltyConfig.objects.get(property=booking.room.property)
    except LoyaltyConfig.DoesNotExist:
        config = None

    first_booking_pts = getattr(config, 'first_booking_pts', 100)
    pts_per_night = getattr(config, 'pts_per_night', 20)
    monthly_repeat_multiplier = float(getattr(config, 'monthly_repeat_multiplier', 1.5))

    prior_confirmed = Booking.objects.filter(
        user=booking.user,
        status='confirmed',
    ).exclude(pk=booking_pk).count()

    nights = booking.num_nights
    base = first_booking_pts if prior_confirmed == 0 else nights * pts_per_night

    multiplier = 1.0
    # Monthly repeat multiplier (≥2 bookings this calendar month)
    month_start = booking.created_at.replace(day=1, hour=0, minute=0, second=0, microsecond=0)
    month_bookings = Booking.objects.filter(
        user=booking.user,
        status='confirmed',
        created_at__gte=month_start,
    ).exclude(pk=booking_pk).count()
    if month_bookings >= 1:
        multiplier = max(multiplier, monthly_repeat_multiplier)

    # Active campaign covering check_in date
    campaigns = CampaignRule.objects.filter(
        is_active=True,
        start_date__lte=booking.check_in,
        end_date__gte=booking.check_in,
    )
    if campaigns.exists():
        best = max(float(c.multiplier) for c in campaigns)
        multiplier = max(multiplier, best)

    final = int(base * multiplier)

    LoyaltyLedger.objects.create(
        user=booking.user,
        delta=final,
        reason='BOOKING_CONFIRMED',
        booking=booking,
    )

    profile, _ = UserProfile.objects.get_or_create(user=booking.user)
    profile.loyalty_points = (profile.loyalty_points or 0) + final
    update_tier(profile)
    profile.save()


def update_tier(profile):
    from .models import LoyaltyTier
    tiers = LoyaltyTier.objects.order_by('min_pts')
    earned_tier = None

            earned_tier = tier
    if earned_tier and profile.loyalty_tier != earned_tier:
```

```python
def loyalty_context(request):
    if not request.user.is_authenticated:
        return {}
    try:
        from .models import LoyaltyTier
        profile = getattr(request.user, 'userprofile', None)
        if not profile:
            return {}

        current_tier = profile.loyalty_tier
        tiers = list(LoyaltyTier.objects.order_by('min_pts'))
        next_tier = None
        progress_pct = 0
        for i, tier in enumerate(tiers):
            if current_tier and tier.pk == current_tier.pk and i + 1 < len(tiers):
                next_tier = tiers[i + 1]
                span = next_tier.min_pts - tier.min_pts
                progress_pct = int(((points - tier.min_pts) / span) * 100) if span > 0 else 100
                break
        return {
            'loyalty_points': points,
            'loyalty_tier': current_tier,
            'loyalty_tier_next': next_tier,
            'loyalty_progress_pct': min(progress_pct, 100),
        }
    except Exception:
        return {}
```
